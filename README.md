# Ternary-operator.java

/*Ternary Operators */

import java.util.*;

class Ternary{

public static void main(String[] args){

Scanner sc = new Scanner(System.in);

System.out.println("Enter Value of a :");

int a = sc.nextInt();

System.out.println("Enter value of b :");

int b = sc.nextInt();

int res=(a>b)?a:b;

System.out.println("Result="+res);

}

}
